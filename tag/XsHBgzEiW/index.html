容器和迭代器

容器的概念很好理解 
所谓迭代器就是一种检查容器内元素并遍历元素的数据类型
初学写的一个小案例：

#include "stdafx.h"
#include
#include
#include"vector"   //使用容器必须加上此头文件

using namespace std;

void main11()
{
vector v1;         //v1是一个容器
//往容器里存入数据
v1.push_back(1);
v1.push_back(3);
v1.push_back(5);
v1.push_back(7);
v1.push_back(9);

cout << "容器v1的长度：" << v1.size() << endl;    //输出容器的长度
cout <<"容器的第三个元素："<< v1[3] << endl; //输出容器的第三个元素
//vector ::iterator iter; //定义一个自己类型的迭代器  iter是名字
//定义一个名为iter的变量，它的数据类型是由vector定义的iterator类型。

for (vector ::iterator iter = v1.begin(); iter != v1.end(); iter++)
{
	cout << *iter << "  "; //遍历输出
}
	cout << endl;
}

int main()
{
main11();
    return 0;
}
